
CODE SEGMENT
ASSUME CS:CODE,DS:CODE,SS:CODE,ES:CODE
PA      EQU 01H
PB      EQU 05H
PC      EQU 09H
CO      EQU 0DH


ORG 0000H

MOV AL,10001000B
OUT CO,AL


BEGIN:

LEA SI,TABLE

START:
   IN AL,PC
   AND AL,10000000B
   CMP AL,00H
   JZ SHOW
   JMP START



SHOW:
      MOV AL,[SI]
      OUT PA,AL
      CALL DELAY
      CMP AL,00H
      JZ  BEGIN
      INC SI
      JMP START



TABLE:
      DB 3FH
      DB 06H
      DB 5BH
      DB 4FH
      DB 66H
      DB 6DH
      DB 7DH
      DB 07H
      DB 7FH
      DB 6FH
      DB 71H   ;F
      DB 00H

DELAY:
MOV CX,29412D
L1:LOOP L1
RET

CODE ENDS
END
