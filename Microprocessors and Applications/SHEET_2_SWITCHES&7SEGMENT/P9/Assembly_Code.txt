CODE SEGMENT
    ASSUME CS:CODE,DS:CODE,SS:CODE,ES:CODE
    PB1 EQU 02H
    PB2 EQU 0AH
    PCON1 EQU 06H
    PCON2 EQU 0EH
    MOV AL,80H
    OUT PCON1 ,AL
    OUT PCON2 ,AL


START:
     MOV AL,00H
     OUT PB1,AL
     OUT PB2,AL
     MOV DX ,10D
LOOP1:
    MOV AL,0FFH
    OUT PB1,AL
    CALL DELAY
    MOV AL,00H
    OUT PB1 ,AL
    CALL DELAY
    DEC DX
    JNZ LOOP1

LEA SI,TABLE
SHOW:
    MOV AL ,[SI]
    OUT PB2,AL
    CALL DELAY
    INC SI
    CMP AL,6FH
    JNZ SHOW


INT 3


TABLE:
DB 3FH
DB 06H
DB 5BH
DB 4FH
DB 66H
DB 6DH
DB 7DH
DB 07H
DB 7FH
DB 6FH



DELAY:
MOV CX,58824D
L1:
LOOP L1
RET
CODE ENDS
END
