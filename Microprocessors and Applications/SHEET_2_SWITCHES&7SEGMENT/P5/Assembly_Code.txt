CODE SEGMENT
ASSUME CS:CODE , DS:CODE ,SS:CODE ,ES:CODE
PA     EQU 01H
PB     EQU 05H
PC     EQU 09H
CO     EQU 0DH

ORG 0000H

MOV AL,10001000B
OUT CO,AL

SHOW_ODD :
LEA SI,TABLE_ODD

DOWN:
IN AL,PC
AND AL,10000000B
CMP AL,00H
jz SHOW_EVEN
MOV AL ,[SI+5]
OUT PA,AL
CALL DELAY
CMP AL,00H
JZ SHOW_ODD
DEC SI
JMP DOWN


SHOW_EVEN:
LEA SI,TABLE_EVEN
UP:
IN AL,PC
AND AL,10000000B
CMP AL,00H
JNZ SHOW_ODD
MOV AL,[SI]
OUT PA,AL
CALL DELAY
CMP AL,00H
JZ SHOW_EVEN
INC SI
JMP UP



TABLE_EVEN:
DB 3FH ;0
DB 5BH ;2
DB 66H ;4
DB 7DH ;6
DB 7FH ;8
DB 00H ;NOTHING TO KNOW THE END OF COUNTER

TABLE_ODD:
DB 00H ;NOTHING TO KNOW THE END OF COUNTER
DB 06H ;1
DB 4FH ;3
DB 6DH ;5
DB 07H ;7
DB 6FH ;9



DELAY:
MOV CX,58824D
L1: LOOP L1
RET

CODE ENDS
END
