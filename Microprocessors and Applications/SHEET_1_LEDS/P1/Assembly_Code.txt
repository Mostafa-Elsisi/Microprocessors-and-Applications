
CODE SEGMENT
ASSUME CS:CODE , DS:CODE ,SS:CODE ,ES:CODE

PA     EQU 00H
PB     EQU 04H
PC     EQU 08H
PCON   EQU 0CH

ORG 0000H
MOV AL , 10000000B
OUT PCON , AL


START:
MOV DL , 5D

NEXT:
MOV AL,00H
OUT PA,AL
OUT PB,AL
CALL DELAY1
MOV AL,0FFH
OUT PA,AL
OUT PB,AL
CALL DELAY2
DEC DL
JNZ NEXT

INT 3

DELAY1:
MOV CX,23529D
L1: LOOP L1
RET
DELAY2:
MOV CX,47058
L2: LOOP L2
RET

CODE ENDS
END
